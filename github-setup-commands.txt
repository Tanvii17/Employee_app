# Initial Setup [First time github setup]
1.  git init
2.  git add .
3.  git commit -m "message"
4.  git remote add origin https://github.com/Tanvii17/Employee-management.git
5.  git remote set-url origin https://Tanvii17:<Token>@github.com/Tanvii17/Employee-management.git
6.  git push --set-upstream origin master

# [From next time to push modified changes]
1. git pull
2. git add .
3. git commit -m "added new commands"
4. git push

# To see new changes

1. git status
